Formal assumptions tests

# These tests require the residuals for the best model; use preds_full CSV
if (!is.null(ms)) {
  best <- df_models %>% arrange(cv_rmse_mean) %>% slice(1) %>% pull(model)
  preds_path <- path(outputs_dir, paste0(best, "_preds_full.csv"))
  if (file_exists(preds_path)) {
    preds <- read_csv(preds_path, show_col_types = FALSE) %>% mutate(resid = y_true - y_pred)
    # Shapiro-Wilk test for normality of residuals (note: sensitive for large N)
    if (nrow(preds) <= 5000) {
      sh <- shapiro.test(preds$resid)
      cat("Shapiro-Wilk p-value:", round(sh$p.value,4), "\n")
      cat("Interpretation: p < 0.05 suggests residuals deviate from normality; with large N, small deviations can be significant.\n")
    } else {
      cat("Shapiro-Wilk not run (n > 5000). Use histogram and skew/kurtosis diagnostics instead.\n")
      cat("Residual skew:", round(e1071::skewness(preds$resid, na.rm=TRUE),3), "kurtosis:", round(e1071::kurtosis(preds$resid, na.rm=TRUE),3), "\n")
    }
    # Breusch-Pagan test for heteroskedasticity
    if (requireNamespace("lmtest", quietly=TRUE)) {
      # fit a simple linear model of y_pred -> y_true for BP test
      lm_fit <- lm(y_true ~ y_pred, data=preds)
      bp <- lmtest::bptest(lm_fit)
      cat("Breusch-Pagan p-value:", round(bp$p.value,4), "\n")
      cat("Interpretation: p < 0.05 indicates heteroskedasticity (non-constant variance).\n")
    } else {
      cat("Package 'lmtest' not installed; cannot run Breusch-Pagan test.\n")
    }
    # Durbin-Watson: only applicable if residuals have temporal ordering; state applicability
    cat("Durbin-Watson test: only applicable if data are time-ordered. If no time index exists, this test is not applicable.\n")
  } else {
    cat("Predictions file not found for", best, "; cannot run formal residual tests.\n")
  }
} else {
  cat("No models_summary.json found.\n")
}

Overfitting control summary

  • Nested cross‑validation (outer 5‑fold, inner 3‑fold) was used to estimate generalization performance and tune hyperparameters.
  • Regularization: ElasticNet uses L1/L2 penalties; tuned alpha and l1_ratio.
  • Tree constraints: max_depth and max_features were included in search spaces to limit overfitting.
  • MLP: early stopping / patience implemented in the PyTorch wrapper to avoid overfitting.
