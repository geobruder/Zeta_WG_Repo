## Formal assumptions tests

```{r assumptions, echo=TRUE}
if (!is.null(ms)) {
  best <- df_models %>% arrange(cv_rmse_mean) %>% slice(1) %>% pull(model)
  preds_path <- path(outputs_dir, paste0(best, "_preds_full.csv"))
  if (file_exists(preds_path)) {
    preds <- read_csv(preds_path, show_col_types = FALSE) %>% mutate(resid = y_true - y_pred)
    if (nrow(preds) <= 5000) {
      sh <- shapiro.test(preds$resid)
      cat("Shapiro-Wilk p-value:", round(sh$p.value,4), "\n")
      cat("Interpretation: p < 0.05 suggests residuals deviate from normality; with large N, small deviations can be significant.\n")
    } else {
      cat("Shapiro-Wilk not run (n > 5000). Use histogram and skew/kurtosis diagnostics instead.\n")
      cat("Residual skew:", round(e1071::skewness(preds$resid, na.rm=TRUE),3), "kurtosis:", round(e1071::kurtosis(preds$resid, na.rm=TRUE),3), "\n")
    }
    if (requireNamespace("lmtest", quietly=TRUE)) {
      lm_fit <- lm(y_true ~ y_pred, data=preds)
      bp <- lmtest::bptest(lm_fit)

      cat("Interpretation: p < 0.05 indicates heteroskedasticity (non-constant variance).\n")
```

## Conformal coverage diagnostic

```{r conformal-coverage, echo=TRUE}
if (!is.null(ms)) {
  cov_rows <- map_dfr(names(ms)[names(ms) != "run_info"], function(nm) {
    v <- ms[[nm]]
    q90 <- v$conformal_q_90 %||% NA
    preds_path <- path(outputs_dir, paste0(nm, "_preds_full.csv"))
    if (file_exists(preds_path) && !is.na(q90)) {
      preds <- read_csv(preds_path, show_col_types = FALSE)
      covered <- mean((preds$y_true >= (preds$y_pred - q90)) & (preds$y_true <= (preds$y_pred + q90)), na.rm=TRUE)
      tibble(model = nm, conformal_q_90 = q90, empirical_coverage = round(covered,3))
    } else {
      tibble(model = nm, conformal_q_90 = q90, empirical_coverage = NA_real_)
    }
  })
  cov_rows %>% kable(caption="Conformal q_90 and empirical coverage (held-out predictions)") %>% kable_styling(full_width = FALSE)
  cat("Interpretation: empirical_coverage close to 0.90 indicates conformal interval q_90 provides near-nominal coverage.\n")
} else {
  cat("models_summary.json not found; run nested CV cell first.\n")
}
```

**Figure 3. Conformal q_90 and empirical coverage table.**  
*Description.* Table listing the split‑conformal 90% residual quantile (`q_90`) and the empirical coverage (fraction of held‑out true values within ±q_90) for each model. *Interpretation.* Empirical coverage near 0.90 indicates the conformal interval is calibrated.
